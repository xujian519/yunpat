## 🎯 使用示例

### 完整工作流

```typescript
import { createMemoryLayer, createTokenWindowManager, createContextManager } from '@yunpat/core';

// 1. 初始化记忆层
const memory = await createMemoryLayer({
  databaseUrl: process.env.DATABASE_URL!,
  vectorDimension: 1024, // BGE-M3
});

// 2. 初始化 Token 窗口
const tokenWindow = createTokenWindowManager({
  maxTokens: 4000,
  reservedTokens: 500,
  enableSummary: true,
});

// 3. 初始化上下文管理器
const contextManager = createContextManager({
  systemPrompt: '你是一个专业的专利撰写助手。',
});

// 4. 添加长期记忆
await memory.addMemory({
  type: 'patent',
  content: '专利 CN123456 涉及人工智能技术',
  embedding: await generateEmbedding('专利 CN123456 涉及人工智能技术'),
  metadata: { agent: 'writer', tags: ['AI', '专利'] },
});

// 5. 管理对话历史
const { messages: compressed } = await tokenWindow.slideWindow(dialogueHistory);

// 6. 检索相关记忆
const relevantMemories = await memory.searchMemories(
  queryEmbedding,
  5,
  { types: ['patent'] }
);

// 7. 构建完整上下文
const { context } = await contextManager.buildContext(compressed);

// 8. 调用 LLM
const response = await callLLM(context);
```
