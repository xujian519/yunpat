- **Node.js**: v22.x
- **PostgreSQL**: v14+ (with pgvector)
- **pnpm**: v9.x
- **Python**: v3.9+ (BGE-M3)

## PostgreSQL 部署

### 1. 安装 pgvector 扩展

```bash
# 编译安装 pgvector
cd /tmp
git clone --branch v0.5.1 https://github.com/pgvector/pgvector.git
cd pgvector

# 确保 pg_config 在 PATH 中
export PATH=/usr/local/bin:$PATH

# 编译安装
make
sudo make install
```

### 2. 创建数据库和用户

```bash
# 切换到 postgres 用户
sudo -u postgres psql

# 在 psql 中执行：
CREATE USER yunpat WITH PASSWORD 'yunpat123';
CREATE DATABASE yunpat OWNER yunpat;
GRANT ALL PRIVILEGES ON DATABASE yunpat TO yunpat;
\q
```

### 3. 启用 pgvector 扩展

```bash
# 连接到数据库
psql -U yunpat -d yunpat

# 在 psql 中执行：
CREATE EXTENSION IF NOT EXISTS vector;

# 验证
SELECT * FROM pg_extension WHERE extname = 'vector';
```

### 4. 创建记忆表

```sql
-- 创建 memories 表
CREATE TABLE IF NOT EXISTS memories (
  id SERIAL PRIMARY KEY,
  type VARCHAR(50) NOT NULL,
  content TEXT NOT NULL,
  embedding vector(1024) NOT NULL,
  metadata JSONB,
  is_archived BOOLEAN DEFAULT FALSE,
  created_at TIMESTAMP DEFAULT NOW(),
  updated_at TIMESTAMP DEFAULT NOW()
);

-- 创建 HNSW 索引
CREATE INDEX IF NOT EXISTS memories_embedding_hnsw_idx 
ON memories USING hnsw (embedding vector_cosine_ops) 
WITH (m = 16, ef_construction = 64);

-- 创建其他索引
CREATE INDEX IF NOT EXISTS memories_type_idx ON memories(type);
CREATE INDEX IF NOT EXISTS memories_created_at_idx ON memories(created_at DESC);
CREATE INDEX IF NOT EXISTS memories_is_archived_idx ON memories(is_archived);
```

## BGE-M3 服务部署

### 1. 安装 Python 依赖

```bash
# 创建虚拟环境
python3.9 -m venv bge-env
source bge-env/bin/activate

# 安装依赖
pip install bge-m3 mlx numpy
```

### 2. 启动 BGE-M3 服务

```bash
# 方式 1: 使用内置服务器（开发）
python -m bge_m3.server --port 8009

# 方式 2: 使用 gunicorn（生产）
pip install gunicorn
gunicorn -w 4 -b 0.0.0.0:8009 bge_m3.server:app
```

### 3. 验证服务

```bash
# 健康检查
curl http://localhost:8009/health

# 测试嵌入
curl -X POST http://localhost:8009/embed \
  -H "Content-Type: application/json" \
  -d '{"texts": ["你好世界"]}'
```

### 4. 配置 API Key（可选）

```bash
# 设置 API Key
export BGE_API_KEY=your-api-key-here

# 或在代码中配置
const bge = createBGEM3Client({
  apiKey: process.env.BGE_API_KEY,
});
```

## Docker 部署

### 1. 使用 Docker Compose

创建 `docker-compose.yml`:

```yaml
version: '3.8'

services:
  postgres:
    image: pgvector/pgvector:pg14
    environment:
      POSTGRES_USER: yunpat
      POSTGRES_PASSWORD: yunpat123
      POSTGRES_DB: yunpat
    ports:
      - "5432:5432"
    volumes:
      - postgres_data:/var/lib/postgresql/data
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U yunpat"]
      interval: 10s
      timeout: 5s
      retries: 5

  bge-m3:
    build:
      context: .
      dockerfile: docker/bge-m3/Dockerfile
    ports:
      - "8009:8009"
    environment:
      - BGE_API_KEY=${BGE_API_KEY}
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost:8009/health"]
      interval: 30s
      timeout: 10s
      retries: 3

  app:
    build:
      context: .
      dockerfile: Dockerfile
    environment:
      - DATABASE_URL=postgres://yunpat:yunpat123@postgres:5432/yunpat
      - BGE_BASE_URL=http://bge-m3:8009
      - BGE_API_KEY=${BGE_API_KEY}
    depends_on:
      - postgres
      - bge-m3
    ports:
      - "3000:3000"

volumes:
  postgres_data:
```

### 2. 启动服务

```bash
# 构建并启动
docker-compose up -d

# 查看日志
docker-compose logs -f

# 停止服务
docker-compose down
```

## 生产环境配置

### 1. PostgreSQL 优化

编辑 `postgresql.conf`:

```conf
# 内存配置
shared_buffers = 256MB
effective_cache_size = 1GB
maintenance_work_mem = 64MB

# 连接配置
max_connections = 100

# 查询优化
random_page_cost = 1.1
effective_io_concurrency = 200

# WAL 配置
wal_buffers = 16MB
checkpoint_completion_target = 0.9
```

### 2. 应用性能优化

```typescript
// 使用连接池
import postgres from 'postgres';

const client = postgres(process.env.DATABASE_URL, {
  max: 10,  // 连接池大小
  idle_timeout: 20,
  connect_timeout: 10,
});

// 使用 HNSW 索引优化
const vectorStore = new PostgresVectorStore({
  databaseUrl: process.env.DATABASE_URL,
  vectorDimension: 1024,
  hnswM: 16,
  hnswEfConstruction: 64,
});
```

### 3. 安全配置

```bash
# 设置文件权限
chmod 600 .env

# 使用环境变量
export DATABASE_URL="postgres://user:pass@host:5432/db"

# 配置防火墙
sudo ufw allow from 203.0.113.0 to any port 5432
```

## 监控与日志

### 1. 应用监控

```typescript
import { PostgresVectorStore } from '@yunpat/core';

async function monitor() {
  const vectorStore = new PostgresVectorStore({
    databaseUrl: process.env.DATABASE_URL,
  });

  // 获取统计信息
  const stats = await vectorStore.getStats();
  console.log(`总记忆数: ${stats.totalMemories}`);
  console.log(`归档记忆数: ${stats.archivedMemories}`);
  console.log(`类型分布:`, stats.typeDistribution);

  // 定期检查
  setInterval(async () => {
    const stats = await vectorStore.getStats();
    if (stats.totalMemories > 100000) {
      console.warn('记忆数量过多，考虑归档');
    }
  }, 60000); // 每分钟检查
}
```

### 2. 日志配置

```typescript
// 使用 winston 或 pino
import pino from 'pino';

const logger = pino({
  level: process.env.LOG_LEVEL || 'info',
  transport: {
    target: 'pino-pretty',
    options: {
      colorize: true,
    },
  },
});

// 使用日志
logger.info('记忆层已启动');
logger.error('数据库连接失败', { error });
```

### 3. 健康检查端点

```typescript
import express from 'express';

const app = express();

app.get('/health', async (req, res) => {
  try {
    // 检查数据库
    const vectorStore = new PostgresVectorStore({
      databaseUrl: process.env.DATABASE_URL,
    });
    await vectorStore.getStats();

    // 检查 BGE-M3
    const bge = createBGEM3Client();
    const healthy = await bge.healthCheck();

    res.json({
      status: 'ok',
      database: 'ok',
      bge: healthy ? 'ok' : 'error',
    });
  } catch (error) {
    res.status(500).json({
      status: 'error',
      message: error.message,
    });
  }
});

app.listen(3000);
```

## 故障排查

### 常见问题

#### 1. pgvector 扩展未安装

**错误**: `could not open extension control file`

**解决**:
```bash
# 检查扩展是否安装
psql -U yunpat -d yunpat -c "SELECT * FROM pg_extension WHERE extname = 'vector';"

# 如果不存在，安装扩展
psql -U yunpat -d yunpat -c "CREATE EXTENSION vector;"
```

#### 2. BGE-M3 服务连接失败

**错误**: `ECONNREFUSED`

**解决**:
```bash
# 检查服务是否运行
curl http://localhost:8009/health

# 检查防火墙
sudo ufw status

# 重启服务
pkill -f bge-m3
python -m bge_m3.server --port 8009
```

#### 3. 向量维度不匹配

**错误**: `向量维度不匹配：期望 1024，实际 768`

**解决**:
```typescript
// 确保向量维度一致
const bge = createBGEM3Client();
const embedding = await bge.embed('test');
console.log(embedding.length); // 应该是 1024

// 如果不是 1024，检查 BGE-M3 配置
```
